```python
n = int(input())
res = 0
for _ in range(n):
    line = input()
    elems = line.split()
    count = 0
    for e in elems:
        if e == '1':
            count+=1
    if count>=2:
        res+=1
print(res)
```

```python
n, k = map(int,input().split())

a = sorted(a)

    if a[0]>1:
        print(1)
    else:
        print(-1)
elif k<n and a[k-1]==a[k]:
    print(-1)
else:
    print(a[k-1])
```
